-- Operations for filtering lists
 
-- The following are a variety of functions that can be used to
-- construct sublists using a predicate.
--
-- Each function has two forms. The first main variant uses a
-- proof-relevant decidable predicate, while the second variant uses
-- a irrelevant boolean predicate and are suffixed with a `·µá` character,
-- typed as \^b.
--
-- The decidable versions have several advantages: 1) easier to prove
-- properties, 2) better meta-variable inference and 3) most of the rest
-- of the library is set-up to work with decidable predicates. However,
-- in rare cases the boolean versions can be useful, mainly when one
-- wants to minimise dependencies.
--
-- In summary, in most cases you probably want to use the decidable
-- versions over the boolean versions, e.g. use `takeWhile (_‚â§? 10) xs`
-- rather than `takeWhile·µá (_‚â§·µá 10) xs`.
 
takeWhile : ‚àÄ {P : Pred A p} ‚Üí Decidable P ‚Üí List A ‚Üí List A
takeWhile P? []       = []
takeWhile P? (x ‚à∑ xs) with does (P? x)
... | true  = x ‚à∑ takeWhile P? xs
... | false = []
 
takeWhile·µá : (A ‚Üí Bool) ‚Üí List A ‚Üí List A
takeWhile·µá p = takeWhile (T? ‚àò p)
 
dropWhile : ‚àÄ {P : Pred A p} ‚Üí Decidable P ‚Üí List A ‚Üí List A
dropWhile P? []       = []
dropWhile P? (x ‚à∑ xs) with does (P? x)
... | true  = dropWhile P? xs
... | false = x ‚à∑ xs
 
dropWhile·µá : (A ‚Üí Bool) ‚Üí List A ‚Üí List A
dropWhile·µá p = dropWhile (T? ‚àò p)
 
filter : ‚àÄ {P : Pred A p} ‚Üí Decidable P ‚Üí List A ‚Üí List A
filter P? [] = []
filter P? (x ‚à∑ xs) with does (P? x)
... | false = filter P? xs
... | true  = x ‚à∑ filter P? xs
 
filter·µá : (A ‚Üí Bool) ‚Üí List A ‚Üí List A
filter·µá p = filter (T? ‚àò p)
 
partition : ‚àÄ {P : Pred A p} ‚Üí Decidable P ‚Üí List A ‚Üí (List A √ó List A)
partition P? []       = ([] , [])
partition P? (x ‚à∑ xs) with does (P? x) | partition P? xs
... | true  | (ys , zs) = (x ‚à∑ ys , zs)
... | false | (ys , zs) = (ys , x ‚à∑ zs)
 
partition·µá : (A ‚Üí Bool) ‚Üí List A ‚Üí List A √ó List A
partition·µá p = partition (T? ‚àò p)
 
span : ‚àÄ {P : Pred A p} ‚Üí Decidable P ‚Üí List A ‚Üí (List A √ó List A)

span·µá : (A ‚Üí Bool) ‚Üí List A ‚Üí List A √ó List A
span·µá p = span (T? ‚àò p)
 
break : ‚àÄ {P : Pred A p} ‚Üí Decidable P ‚Üí List A ‚Üí (List A √ó List A)
break P? = span (¬¨? ‚àò P?)
 
break·µá : (A ‚Üí Bool) ‚Üí List A ‚Üí List A √ó List A
break·µá p = break (T? ‚àò p)
 
-- The predicate `P` represents the notion of newline character for the
-- type `A`. It is used to split the input list into a list of lines.
-- Some lines may be empty if the input contains at least two
-- consecutive newline characters.
linesBy : ‚àÄ {P : Pred A p} ‚Üí Decidable P ‚Üí List A ‚Üí List (List A)
linesBy {A = A} P? = go nothing where
 
  go : Maybe (List A) ‚Üí List A ‚Üí List (List A)
  go acc []       = maybe‚Ä≤ ([_] ‚àò‚Ä≤ reverse) [] acc
  go acc (c ‚à∑ cs) = if does (P? c)
    then reverse acc‚Ä≤ ‚à∑ go nothing cs
    else go (just (c ‚à∑ acc‚Ä≤)) cs
    where acc‚Ä≤ = Maybe.fromMaybe [] acc

linesBy·µá p = linesBy (T? ‚àò p)
 
-- The predicate `P` represents the notion of space character for the
-- type `A`. It is used to split the input list into a list of words.
-- All the words are non empty and the output does not contain any space
-- characters.
wordsBy : ‚àÄ {P : Pred A p} ‚Üí Decidable P ‚Üí List A ‚Üí List (List A)

  cons : List A ‚Üí List (List A) ‚Üí List (List A)
  cons [] ass = ass
  cons as ass = reverse as ‚à∑ ass
 
  go : List A ‚Üí List A ‚Üí List (List A)

  go acc (c ‚à∑ cs) = if does (P? c)
    then cons acc (go [] cs)
    else go (c ‚à∑ acc) cs

wordsBy·µá p = wordsBy (T? ‚àò p)
