    a b c : Level


    A : Set a

    C : Set c

-- Definition




open import Agda.Builtin.Maybe public


  using (Maybe; just; nothing)




------------------------------------------------------------------------


-- Some operations




boolToMaybe : Bool ‚Üí Maybe ‚ä§


boolToMaybe true  = just _


boolToMaybe false = nothing




is-just : Maybe A ‚Üí Bool


is-just (just _) = true


is-just nothing  = false




is-nothing : Maybe A ‚Üí Bool


is-nothing = not ‚àò is-just




-- A dependent eliminator.




maybe : ‚àÄ {A : Set a} {B : Maybe A ‚Üí Set b} ‚Üí


        ((x : A) ‚Üí B (just x)) ‚Üí B nothing ‚Üí (x : Maybe A) ‚Üí B x


maybe j n (just x) = j x


maybe j n nothing  = n




-- A non-dependent eliminator.




maybe‚Ä≤ : (A ‚Üí B) ‚Üí B ‚Üí Maybe A ‚Üí B


maybe‚Ä≤ = maybe




-- A defaulting mechanism




fromMaybe : A ‚Üí Maybe A ‚Üí A


fromMaybe = maybe‚Ä≤ id




-- A safe variant of "fromJust". If the value is nothing, then the


-- return type is the unit type.




module _ {a} {A : Set a} where




  From-just : Maybe A ‚Üí Set a


  From-just (just _) = A


  From-just nothing  = Lift a ‚ä§




  from-just : (x : Maybe A) ‚Üí From-just x


  from-just (just x) = x


  from-just nothing  = _




-- Functoriality: map




map : (A ‚Üí B) ‚Üí Maybe A ‚Üí Maybe B


map f = maybe (just ‚àò f) nothing




-- Applicative: ap




ap : Maybe (A ‚Üí B) ‚Üí Maybe A ‚Üí Maybe B


ap nothing  = const nothing


ap (just f) = map f




-- Monad: bind




infixl 1 _>>=_


_>>=_ : Maybe A ‚Üí (A ‚Üí Maybe B) ‚Üí Maybe B


nothing >>= f = nothing


just a  >>= f = f a




-- Alternative: <‚à£>




infixr 6 _<‚à£>_


_<‚à£>_ : Maybe A ‚Üí Maybe A ‚Üí Maybe A


just x  <‚à£> my = just x


nothing <‚à£> my = my




-- Just when the boolean is true




when : Bool ‚Üí A ‚Üí Maybe A


when b c = map (const c) (boolToMaybe b)




------------------------------------------------------------------------


-- Aligning and zipping




alignWith : (These A B ‚Üí C) ‚Üí Maybe A ‚Üí Maybe B ‚Üí Maybe C


alignWith f (just a) (just b) = just (f (these a b))


alignWith f (just a) nothing  = just (f (this a))


alignWith f nothing  (just b) = just (f (that b))


alignWith f nothing  nothing  = nothing




zipWith : (A ‚Üí B ‚Üí C) ‚Üí Maybe A ‚Üí Maybe B ‚Üí Maybe C


zipWith f (just a) (just b) = just (f a b)


zipWith _ _        _        = nothing




align : Maybe A ‚Üí Maybe B ‚Üí Maybe (These A B)


align = alignWith id




zip : Maybe A ‚Üí Maybe B ‚Üí Maybe (A √ó B)


zip = zipWith _,_




------------------------------------------------------------------------


-- Injections.




thisM : A ‚Üí Maybe B ‚Üí These A B


thisM a = maybe‚Ä≤ (these a) (this a)




thatM : Maybe A ‚Üí B ‚Üí These A B


thatM = maybe‚Ä≤ these that
