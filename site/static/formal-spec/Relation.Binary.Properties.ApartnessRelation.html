¬#-isEquivalence : Reflexive _≈_ → IsApartnessRelation _≈_ _#_ →
                   IsEquivalence (¬_ ∘₂ _#_)
¬#-isEquivalence re apart = record
  { refl = irrefl re
  ; sym = λ {a} {b} → sym⇒¬-sym sym {a} {b}
  ; trans = cotrans⇒¬-trans cotrans
  } where open IsApartnessRelation apart
