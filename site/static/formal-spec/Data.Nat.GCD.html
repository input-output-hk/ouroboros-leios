-- Definition
 
-- Calculated via Euclid's algorithm. In order to show progress,
-- avoiding the initial step where the first argument may increase, it
-- is necessary to first define a version `gcd‚Ä≤` which assumes that the
-- second argument is strictly smaller than the first. The full `gcd`
-- function then compares the two arguments and applies `gcd‚Ä≤`
-- accordingly.

gcd‚Ä≤ m n@(suc _) (acc rec) n<m = gcd‚Ä≤ n (m % n) (rec n<m) (m%n<n m n)

-- A formal specification of GCD
 
module GCD where
 
  -- Specification of the greatest common divisor (gcd) of two natural
  -- numbers.
 
  record GCD (m n gcd : ‚Ñï) : Set where
    constructor is
    field
      -- The gcd is a common divisor.
      commonDivisor : gcd ‚à£ m √ó gcd ‚à£ n
 
      -- All common divisors divide the gcd, i.e. the gcd is the
      -- greatest common divisor according to the partial order _‚à£_.
      greatest : ‚àÄ {d} ‚Üí d ‚à£ m √ó d ‚à£ n ‚Üí d ‚à£ gcd
 
    gcd‚à£m : gcd ‚à£ m
    gcd‚à£m = proj‚ÇÅ commonDivisor
 
    gcd‚à£n : gcd ‚à£ n
    gcd‚à£n = proj‚ÇÇ commonDivisor
 
  open GCD public
 
 
  -- The gcd is unique.
 
  unique : ‚àÄ {d‚ÇÅ d‚ÇÇ m n} ‚Üí GCD m n d‚ÇÅ ‚Üí GCD m n d‚ÇÇ ‚Üí d‚ÇÅ ‚â° d‚ÇÇ
  unique d‚ÇÅ d‚ÇÇ = ‚à£-antisym (GCD.greatest d‚ÇÇ (GCD.commonDivisor d‚ÇÅ))
                           (GCD.greatest d‚ÇÅ (GCD.commonDivisor d‚ÇÇ))
 
  -- The gcd relation is "symmetric".
 
  sym : ‚àÄ {d m n} ‚Üí GCD m n d ‚Üí GCD n m d
  sym g = is (swap $ GCD.commonDivisor g) (GCD.greatest g ‚àò swap)
 
  -- The gcd relation is "reflexive".
 
  refl : ‚àÄ {n} ‚Üí GCD n n n
  refl = is (‚à£-refl , ‚à£-refl) proj‚ÇÅ
 
  -- The GCD of 0 and n is n.
 
  base : ‚àÄ {n} ‚Üí GCD 0 n n
  base {n} = is (n ‚à£0 , ‚à£-refl) proj‚ÇÇ
 
  -- If d is the gcd of n and k, then it is also the gcd of n and
  -- n + k.
 
  step : ‚àÄ {n k d} ‚Üí GCD n k d ‚Üí GCD n (n + k) d
  step {n} {k} {d} g with d‚ÇÅ , d‚ÇÇ ‚Üê GCD.commonDivisor g

    where
    greatest‚Ä≤ : ‚àÄ {d‚Ä≤} ‚Üí d‚Ä≤ ‚à£ n √ó d‚Ä≤ ‚à£ n + k ‚Üí d‚Ä≤ ‚à£ d
    greatest‚Ä≤ (d‚ÇÅ , d‚ÇÇ) = GCD.greatest g (d‚ÇÅ , ‚à£m+n‚à£m‚áí‚à£n d‚ÇÇ d‚ÇÅ)
