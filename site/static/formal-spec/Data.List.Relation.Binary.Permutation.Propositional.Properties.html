-- merge
 
module _ {‚Ñì} {R : Rel A ‚Ñì} (R? : Decidable R) where

-- filter
 
filter-‚Ü≠ : ‚àÄ {p} {P : Pred A p} (P? : Pred.Decidable P) ‚Üí xs ‚Ü≠ ys ‚Üí filter P? xs ‚Ü≠ filter P? ys

-- catMaybes
